# SkillVerse — Database Schema

> **Version**: 1.0 — MVP  
> **Database**: Supabase (PostgreSQL 15)  
> **Last Updated**: 2026-07-17  

---

## Overview

SkillVerse uses **8 core tables**. All tables use UUIDs as primary keys, reference `auth.users` managed by Supabase Auth, and have Row Level Security (RLS) enabled so users can only access their own data.

---

## Entity Relationship Diagram

```
auth.users (Supabase managed)
    │
    ├─── learner_profiles (1:1)
    │         │
    │         ├─── roadmaps (1:N)
    │         │         │
    │         │         └─── milestones (1:N)
    │         │                   │
    │         │                   ├─── daily_tasks (1:N)
    │         │                   ├─── resources (1:N)
    │         │                   ├─── projects (1:1)
    │         │                   └─── quizzes (1:N)
    │         │
    │         └─── progress (1:1 per roadmap)
```

## Table 1: `learner_profiles`

Stores the user's learning goal, preferences, and AI-analyzed goal breakdown.

```sql
CREATE TABLE learner_profiles (
  id                UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id           UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  
  -- Raw user input
  raw_goal          TEXT NOT NULL CHECK (char_length(raw_goal) BETWEEN 10 AND 500),
  skill_level       TEXT CHECK (skill_level IN ('beginner', 'intermediate', 'advanced')),
  hours_per_week    SMALLINT CHECK (hours_per_week BETWEEN 1 AND 40),
  deadline_weeks    SMALLINT CHECK (deadline_weeks BETWEEN 1 AND 52),
  
  -- Goal Analyzer Agent output
  domain            TEXT,
  sub_topics        JSONB DEFAULT '[]',          -- string[]
  estimated_weeks   SMALLINT,
  difficulty        TEXT CHECK (difficulty IN ('beginner', 'intermediate', 'advanced')),
  confidence_score  NUMERIC(3,2),
  goal_summary      TEXT,
  
  -- Metadata
  created_at        TIMESTAMPTZ DEFAULT NOW(),
  updated_at        TIMESTAMPTZ DEFAULT NOW(),
  
  CONSTRAINT one_profile_per_user UNIQUE (user_id)
);

-- RLS
ALTER TABLE learner_profiles ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users manage own profile" ON learner_profiles
  FOR ALL USING (auth.uid() = user_id);
```

**Indexes**:
```sql
CREATE INDEX idx_learner_profiles_user_id ON learner_profiles(user_id);
```

## Table 2: `roadmaps`

The generated learning roadmap for a user's goal.

```sql
CREATE TABLE roadmaps (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id         UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  profile_id      UUID NOT NULL REFERENCES learner_profiles(id) ON DELETE CASCADE,
  
  title           TEXT NOT NULL,               -- e.g., "Backend Developer in 12 Weeks"
  total_weeks     SMALLINT NOT NULL,
  start_date      DATE DEFAULT CURRENT_DATE,
  end_date        DATE,                        -- computed: start_date + (total_weeks * 7)
  status          TEXT DEFAULT 'active' CHECK (status IN ('active', 'completed', 'paused')),
  
  created_at      TIMESTAMPTZ DEFAULT NOW(),
  updated_at      TIMESTAMPTZ DEFAULT NOW()
);

ALTER TABLE roadmaps ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users manage own roadmaps" ON roadmaps
  FOR ALL USING (auth.uid() = user_id);
```

## Table 3: `milestones`

Individual curriculum milestones within a roadmap.

```sql
CREATE TABLE milestones (
  id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id             UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  roadmap_id          UUID NOT NULL REFERENCES roadmaps(id) ON DELETE CASCADE,
  
  sequence_number     SMALLINT NOT NULL,        -- 1-based ordering
  title               TEXT NOT NULL,
  description         TEXT NOT NULL,
  estimated_days      SMALLINT NOT NULL,
  topics              JSONB DEFAULT '[]',        -- string[]
  learning_objectives JSONB DEFAULT '[]',        -- string[]
  
  status              TEXT DEFAULT 'not_started' 
                      CHECK (status IN ('not_started', 'in_progress', 'completed')),
  started_at          TIMESTAMPTZ,
  completed_at        TIMESTAMPTZ,
  
  created_at          TIMESTAMPTZ DEFAULT NOW(),
  updated_at          TIMESTAMPTZ DEFAULT NOW(),
  
  CONSTRAINT unique_milestone_sequence UNIQUE (roadmap_id, sequence_number)
);

ALTER TABLE milestones ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users manage own milestones" ON milestones
  FOR ALL USING (auth.uid() = user_id);
```

**Indexes**:
```sql
CREATE INDEX idx_milestones_roadmap_id ON milestones(roadmap_id);
CREATE INDEX idx_milestones_sequence ON milestones(roadmap_id, sequence_number);
```

## Table 4: `daily_tasks`

Day-by-day tasks generated by the Planner Agent.

```sql
CREATE TABLE daily_tasks (
  id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id             UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  roadmap_id          UUID NOT NULL REFERENCES roadmaps(id) ON DELETE CASCADE,
  milestone_id        UUID NOT NULL REFERENCES milestones(id) ON DELETE CASCADE,
  
  day_number          SMALLINT NOT NULL,         -- Day 1, 2, 3...
  task_date           DATE NOT NULL,
  title               TEXT NOT NULL,
  description         TEXT,
  task_type           TEXT NOT NULL 
                      CHECK (task_type IN ('Read', 'Watch', 'Practice', 'Build', 'Quiz')),
  estimated_minutes   SMALLINT NOT NULL CHECK (estimated_minutes > 0),
  
  -- Completion tracking
  is_completed        BOOLEAN DEFAULT FALSE,
  completed_at        TIMESTAMPTZ,
  
  -- Optional link to a resource
  resource_id         UUID REFERENCES resources(id) ON DELETE SET NULL,
  
  created_at          TIMESTAMPTZ DEFAULT NOW()
);

ALTER TABLE daily_tasks ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users manage own tasks" ON daily_tasks
  FOR ALL USING (auth.uid() = user_id);
```

**Indexes**:
```sql
CREATE INDEX idx_daily_tasks_user_date ON daily_tasks(user_id, task_date);
CREATE INDEX idx_daily_tasks_roadmap ON daily_tasks(roadmap_id);
CREATE INDEX idx_daily_tasks_milestone ON daily_tasks(milestone_id);
```

## Table 6: `projects`

Capstone projects generated by the Project Mentor Agent.

```sql
CREATE TABLE projects (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id         UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  milestone_id    UUID NOT NULL REFERENCES milestones(id) ON DELETE CASCADE,
  
  title           TEXT NOT NULL,
  description     TEXT NOT NULL,
  objective       TEXT NOT NULL,
  tech_stack      JSONB DEFAULT '[]',           -- string[]
  steps           JSONB DEFAULT '[]',           -- ProjectStep[]
  deliverable     TEXT NOT NULL,
  estimated_hours SMALLINT,
  
  status          TEXT DEFAULT 'not_started'
                  CHECK (status IN ('not_started', 'in_progress', 'completed')),
  started_at      TIMESTAMPTZ,
  completed_at    TIMESTAMPTZ,
  
  -- Q&A conversation history stored as JSONB
  qa_history      JSONB DEFAULT '[]',           -- { role, message, timestamp }[]
  
  created_at      TIMESTAMPTZ DEFAULT NOW(),
  updated_at      TIMESTAMPTZ DEFAULT NOW(),
  
  CONSTRAINT one_project_per_milestone UNIQUE (milestone_id)
);

ALTER TABLE projects ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users manage own projects" ON projects
  FOR ALL USING (auth.uid() = user_id);
```

## Table 8: `progress`

Aggregated progress tracking per roadmap.

```sql
CREATE TABLE progress (
  id                    UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id               UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  roadmap_id            UUID NOT NULL REFERENCES roadmaps(id) ON DELETE CASCADE,
  
  -- Aggregate counts (updated via triggers or application logic)
  total_tasks           SMALLINT DEFAULT 0,
  completed_tasks       SMALLINT DEFAULT 0,
  completion_percentage NUMERIC(5,2) DEFAULT 0.0,
  
  -- Streak tracking
  current_streak        SMALLINT DEFAULT 0,
  longest_streak        SMALLINT DEFAULT 0,
  last_activity_date    DATE,
  
  -- Milestone counts
  total_milestones      SMALLINT DEFAULT 0,
  completed_milestones  SMALLINT DEFAULT 0,
  
  -- Activity log for heatmap (daily summary)
  activity_log          JSONB DEFAULT '[]',  -- { date, tasks_completed, minutes_spent }[]
  
  created_at            TIMESTAMPTZ DEFAULT NOW(),
  updated_at            TIMESTAMPTZ DEFAULT NOW(),
  
  CONSTRAINT one_progress_per_roadmap UNIQUE (roadmap_id)
);

ALTER TABLE progress ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users manage own progress" ON progress
  FOR ALL USING (auth.uid() = user_id);
```

## Triggers

### Auto-update `updated_at`

```sql
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$ language 'plpgsql';

-- Apply to all tables with updated_at
CREATE TRIGGER update_learner_profiles_updated_at
  BEFORE UPDATE ON learner_profiles
  FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

-- (repeat for roadmaps, milestones, projects, quizzes, progress)
```

### Auto-update Progress on Task Completion

```sql
CREATE OR REPLACE FUNCTION sync_progress_on_task_complete()
RETURNS TRIGGER AS $$
BEGIN
  -- Update progress table when a task is marked complete
  IF NEW.is_completed = TRUE AND OLD.is_completed = FALSE THEN
    UPDATE progress
    SET 
      completed_tasks = completed_tasks + 1,
      completion_percentage = ROUND((completed_tasks + 1)::numeric / total_tasks * 100, 2),
      last_activity_date = CURRENT_DATE,
      updated_at = NOW()
    WHERE roadmap_id = NEW.roadmap_id AND user_id = NEW.user_id;
  END IF;
  RETURN NEW;
END;
$$ language 'plpgsql';

CREATE TRIGGER trigger_progress_on_task_complete
  AFTER UPDATE ON daily_tasks
  FOR EACH ROW EXECUTE FUNCTION sync_progress_on_task_complete();
```
